        function convertir() {
            const cantidad = document.getElementById("cantidad").value;
            const tipo = document.getElementById("conversion").value;
            const resultado = document.getElementById("resultado");

            if (!cantidad || isNaN(cantidad)) {
                resultado.textContent = "‚ö†Ô∏è Ingresa una cantidad v√°lida.";
                return;
            }

            fetch(`http://localhost:8080/PLANTILLA_JERSY/api/conversiones/convertir?monto=${cantidad}&tipo=${tipo}`)
                .then(response => {
                    if (!response.ok) {
                        throw new Error("Error en el servidor");
                    }
                    return response.json();
                })
                .then(data => {
                    resultado.textContent = `üí∞ Resultado: ${data.resultado}`;
                })
                .catch(error => {
                    console.error("Error:", error);
                    resultado.textContent = "‚ùå Error al realizar la conversi√≥n.";
                });
        }
